# jsoup支持库

# 支持库基于jsoup，详细api文档请查询jsoupopen in new window，用法与java类似。

# 解析XML/HTML内容

# 函数名：Jsoup.parse
# 参数列表：
参数类型描述可空
contentStringXML/HTML内容False
# 返回值：
类型描述
Document解析后的Document对象

# 示例：

local html = [[
        <html>
            <head>
                <title>First parse</title>
            </head>
            <body>
                <p id = "content" >Parsed HTML into a doc.</p>
            </body>
        </html>
]]
local doc = Jsoup.parse(html)
local ele = doc:getElementById("content")
print(ele:text())
